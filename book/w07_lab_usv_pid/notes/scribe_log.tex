% ── Pre-run checklist ─────────────────────────────────────────────────────────
\noindent{\bfseries Pre-Run Checklist} (verify before every arming)

\begin{itemize}[leftmargin=1.5em, itemsep=2pt, label=\cb]
  \item Mode set to \textbf{ACRO} before arming
  \item Parameter file saved in Mission Planner \textbf{before} arming (timestamp in filename)
  \item Scribe records time and log number \textbf{at} arming
  \item Same throttle/rudder step fraction used across Run~1 and Run~2
  \item Surge and yaw trials kept in \textbf{separate} log files (disarm between channels)
\end{itemize}
